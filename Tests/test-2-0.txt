                $('#exr-submit').addClass('hide');
            } else {
                $('#exr-next').addClass('hide');
                $('#exr-submit').removeClass('hide');
            }
            if (!$el.data('autoPlayed')) {
                $el.find('.circle-audio').autoPlay();
                $el.data('autoPlayed', 1);
            } else if (e.direction === 'left') {
                $el.prev().stopPlay();
            } else {
                $el.next().stopPlay();
            }
            $(this).exerciseTest('setSubmitParams', 'recent_index', index);
            activeAnswerSheetPanel(index);
        }).on('slide.bs.carousel', function (e) {
            $('.item.active .exercise-view', this).trigger('next.hsk.test');
        }).on('beforeSubmit', function (e) {
            e.submitParams.duration = timeTotal - timeLeft;
        });
        $('#exr-submit').on('click', function (e) {
            if(window.confirm('确定要提交试卷吗?')){
                $('.item.active > .exercise-view', '#exercise-list').trigger('next.hsk.test');
                $('#exercise-list').exerciseTest('submit', true).done(function (data) {
                    window.location.href = '/vi/2/4/report?pid=' + data.exam_id;
                });
            }
        });
        function activeAnswerSheetPanel(index) {
            var $collapse = $('li[data-slide-to=' + index + ']', '#answer-sheets').closest('.collapse');
            if (!$collapse.hasClass('in')) {
                $collapse.prev('.panel-heading').find('[data-toggle=collapse]').trigger('click');
            }
        }
        activeAnswerSheetPanel(0);
        window.setTimeout(function () {
            $('.item.active', '#exercise-list').data('autoPlayed', 1).find('.circle-audio').autoPlay();
        }, 1000);
    
});</script>
